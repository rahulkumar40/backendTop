const generateAvatarInitials = (name) => {
  if (!name) return "??";
  const parts = name.split(" ").filter((part) => part.length > 0);
  if (parts.length === 1) {
    return parts[0].substring(0, 2).toUpperCase();
  }
  return parts
    .map((part) => part[0])
    .join("")
    .substring(0, 2)
    .toUpperCase();
};
